let five	= \s z. s (s (s (s (s z)))) in
let add		= \a b s z. a s (b s z) in
let mul		= \a b s z. a (b s) z in
let ten		= add five five in
let hundred	= mul ten ten in
let thousand	= mul ten hundred in

thousand
